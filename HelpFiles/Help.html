Multi-function Download Session Results Button

Description:

The multi-function download button allows users to download session results from a table. When clicked, it calls the DownloadSessionResults API endpoint to retrieve the session results as a file.

Features:

  • Download session results from the table.
  • Display download progress using a progress bar.

Sample React Code:


import React, { useState } from 'react';
import axios from 'axios';

const DownloadButton = () => {
  const [progress, setProgress] = useState(0);

  const handleDownload = async () => {
    const clientId = 'f89d2d36-7b3b-4e6a-8f43-0951f067fd11';
    const sessionId = '6a9ea5e8-9e2f-4d48-9dc8-45ac9dfd9023';
    const apiUrl = 'https://localhost:5001/AdminApi/DownloadSessionResults';

    const response = await axios.post(apiUrl, { clientId, sessionId }, {
      responseType: 'blob',
      onDownloadProgress: (progressEvent) => {
        const progress = Math.round((progressEvent.loaded / progressEvent.total) * 100);
        setProgress(progress);
      },
    });

    const blob = new Blob([response.data]);
    const downloadUrl = window.URL.createObjectURL(blob);
    const link = document.createElement('a');
    link.href = downloadUrl;
    link.setAttribute('download', 'file_name');
    document.body.appendChild(link);
    link.click();
    setProgress(0);
  };

  return (
    <div>
      <button onClick={handleDownload}>Download File</button>
      {progress > 0 && <progress value={progress} max={100} />}
    </div>
  );
};

export default DownloadButton;

  • Get json value for a location or provider from a live feed data.
  • Display download progress using a progress bar.

Sample React Code:


    import React, { useState } from 'react';
    import axios from 'axios';
    
    const GetInputEntryJsonValue = () => {
      const [jsonValue, setJsonValue] = useState(null);
    
      const handleGetJsonValue = async () => {
        const trackingId = 'f87d4d77-7b3b-4e6a-8f43-0951f067fd11';
        const jsonRestoreType = 'Locations'; // The value will be Locations or Providers.
        const clientId = 'f89d2d36-7b3b-4e6a-8f43-0951f067fd11'; // Provide the client ID if needed
        const apiUrl = `https://localhost:5001/AdminApi/GetSessionInputEntryJsonValue?groupId=${groupId}&jsonRestoreType=${jsonRestoreType}&clientId=${clientId}`;
    
        try {
          const response = await axios.get(apiUrl);
          // Set the JSON value received from the API response
          setJsonValue(response.data);
        } catch (error) {
          console.error('Error occurred while fetching JSON value:', error);
          // Reset JSON value state in case of error
          setJsonValue(null);
        }
      };
    
      return (
        <div>
          <button onClick={handleGetJsonValue}>Get JSON Value</button> 
          <div> 
          {/* Display the JSON value received from the API */}
          <pre> {jsonValue ? JSON.stringify(jsonValue, null, 2) : 'No JSON value retrieved yet'}  </pre>
          </div>          
        </div>
      );
    };
    
    export default GetInputEntryJsonValue;

Import button for locations and providers

Description:

The import button allows users call ImportFromInputData API endpoint.

Features:

  • Display import progress using a progress bar.

Sample React Code:


  import React, { useState } from 'react';

  const ImportDataButton = () => {
    const [isLoading, setIsLoading] = useState(false);
  
    const handleImportData = async () => {
      setIsLoading(true);
  
      try {
        // Generate GUIDs for locations and providers
        const locations = ['1d106db4-8c21-4a05-9a62-d2319e1f9e6c','c1cfda62-ea6b-4d08-a8d9-5b5c07f581ef','92530053-1f14-4ab4-9059-01e021d33d56','87fb18d4-8fc8-4a28-a2a6-77e21fd22921'];
        const providers = ['6fc7f6b9-13f3-4a7d-b45b-8b90a7d8602e','7afdfc16-43ef-4b45-8df5-77725f4918a5','e85d1935-d065-465e-b5a3-2e6847352b25','203a17b5-1007-4a91-af38-5a33064c0977'];
  
        const requestBody = {
          locations: locations,
          providers: providers
        };
  
        const targetEnvironment = 'Staging';
  
        const response = await fetch(`https://localhost:5001/AdminApi/ImportFromInputData?targetEnvironment=${targetEnvironment}`, {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json'
          },
          body: JSON.stringify(requestBody)
        });
  
        if (!response.ok) {
          throw new Error('Failed to import data');
        }
  
        // Assuming the response is JSON
        const data = await response.json();
        console.log('Import successful', data);
      } catch (error) {
        console.error('Error importing data:', error.message);
      } finally {
        setIsLoading(false);
      }
    };
  
    return (
      <div>
      <button onClick={handleImportData} disabled={isLoading}>
        {isLoading ? 'Importing...' : 'Import Data'}
      </button>
      </div>
    );
  };
  
  export default ImportDataButton;
